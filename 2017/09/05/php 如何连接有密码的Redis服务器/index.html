php 如何连接有密码的Redis服务器

Sep 5, 2017
//实例化  
$redis = new Redis();  
//连接服务器  
$redis->connect("localhost");  
//授权  
$redis->auth("passwd");  
//相关操作  
$redis->set("name","father");  
$data = $redis->key("*");  
var_dump($data);
